services:
  web:
    container_name: web
    build:
      context: ./docker/web
      dockerfile: Dockerfile
    environment:
      TZ: "Asia/Tokyo"
    ports:
      - "80:80"
    volumes:
      - ./batch:/udemy/batch
      - ./app:/udemy/app
    # 利用するネットワーク
    networks:
      - php-mysql-networks
    depends_on:
      - db

  db:
    image: mysql:8.0
    container_name: db
    ports:
      - 3306:3306
    environment:
      MYSQL_ROOT_PASSWORD: root
      MYSQL_DATABASE: test_db
      MYSQL_USER: test
      MYSQL_PASSWORD: test
      TZ: "Asia/Tokyo"
    volumes:
      - "udemy_data:/var/lib/mysql"
      - "./docker/db/init:/docker-entrypoint-initdb.d"
      - "./docker/db/my.cnf:/etc/mysql/conf.d/my.cnf"
    restart: always
    # コンテナ間で通信するためのネットワーク（同じに）
    networks:
      - php-mysql-networks

  adminer:
    image: adminer:4.8.1
    container_name: adminer
    ports:
      - 8081:8080
    networks:
      - php-mysql-networks
    depends_on:
      - db

volumes:
  udemy_data:
    driver: local

networks:
  php-mysql-networks:
